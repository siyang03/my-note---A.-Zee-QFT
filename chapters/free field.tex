\chapter{free field}
\begin{itemize}
	\item 考虑如下标量场,
	\begin{equation}
		\mathcal{L} = - \frac{1}{2} (\partial \phi)^2 - V(\phi)
	\end{equation}
	
	\item 含有 source function 的路径积分为,
	\begin{equation}
		Z(J) = \int D\phi e^{i \int d^d x (- \frac{1}{2} (\partial \phi)^2 - V(\phi) + J(x) \phi(x))}
	\end{equation}
	
	\item 当 $V(\phi) = \frac{1}{2} m^2 \phi^2$ 时, 称作 free or Gaussian theory.
\end{itemize}
